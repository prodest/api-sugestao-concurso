import { Injectable } from '@nestjs/common';
import { DatabaseService } from '../database/database.service';
import { PessoaDto } from '../../dto/pessoa.dto';
 
@Injectable()
export class SelecaodtService {
 
    constructor(
        private readonly databaService : DatabaseService, 
    ) {} 
 
    async returnPeopleByOffice(offices){
        const peopleDto: PessoaDto[] = new Array<PessoaDto>(); 
        for(let i = 0; i < offices.length; i++){
            let people = await this.databaService.findByOfficeName(offices[i]);
            let arrayCPF = this.returnArrayCPF(people);
            let personDto = new PessoaDto(offices[i], arrayCPF); 
            //console.log(people);
            
            peopleDto.push(personDto);
        }
        return peopleDto;
        
    }
 
    returnArrayCPF(people){
        let arrayCPF: String[] = new Array<String>();
        for(let i = 0; i < people.length; i++){
            arrayCPF.push(people[i].numerocpf);
        }
        return arrayCPF;
    }
 
}
