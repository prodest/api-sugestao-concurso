import { Injectable, Inject } from '@nestjs/common';
import { Repository } from 'typeorm';
import { Pessoa } from 'src/identidade/core/entities/pessoa.entity';
 
@Injectable()
export class SugestaoBaseService {
  constructor(
    @Inject('PessoaRepositoryToken')
    private readonly pessoaRepository: Repository<Pessoa>,
  ) {}
 
  findByOfficeName(cargo: String): Promise<any> {
    return this.pessoaRepository
      .createQueryBuilder('pessoa')
      .select('numeroCPF')
      .innerJoin('pessoa.habilidade', 'habilidade')
      .where('habilidade.descricao like :nome', { nome: cargo })
      .getRawMany();
  }
}
